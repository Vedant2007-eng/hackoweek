// tracking which menu
var currentMenu = 'chinese';

// italian menu list
var italianMenu = [
    {name:'Margherita Pizza', price:250},
    {name:'Pasta Alfredo', price:220},
    {name:'Lasagna', price:280},
    {name:'Garlic Bread', price:120},
    {name:'Tiramisu', price:180}
];

// function for next page
function nextPage(){
    if(currentMenu == 'chinese'){
        currentMenu = 'italian';
        document.getElementById('menu-title').innerText = 'üçù Italian Food Menu';
        var container = document.getElementById('menu-items');
        container.innerHTML = ''; // clear old

        for(var i=0; i<italianMenu.length; i++){
            var div = document.createElement('div');
            div.className = 'menu-item';

            // making label and span
            var lbl = document.createElement('label');
            var chk = document.createElement('input');
            chk.type = 'checkbox';
            lbl.appendChild(chk);
            lbl.appendChild(document.createTextNode(' ' + italianMenu[i].name));

            var spn = document.createElement('span');
            spn.innerText = '‚Çπ' + italianMenu[i].price;

            div.appendChild(lbl);
            div.appendChild(spn);
            container.appendChild(div);
        }

        // add a back button manually
        var btns = document.querySelector('.buttons');
        if(!document.querySelector('.back')){
            var backBtn = document.createElement('button');
            backBtn.className = 'back';
            backBtn.innerText = 'Back';
            backBtn.onclick = previousPage;
            btns.insertBefore(backBtn, btns.querySelector('.next'));
        }
    }
}

// back to chinese
function previousPage(){
    currentMenu = 'chinese';
    document.getElementById('menu-title').innerText = 'ü•¢ Chinese Food Menu';
    var container = document.getElementById('menu-items');
    container.innerHTML = '';

    var chineseDishes = [

        div.className = 'menu-item';

        div.appendChild(lbl);

    var backBtn = document.querySelector('.back');
    if(backBtn) backBtn.remove();
}

// confirm order
function confirmOrder(){
    var selected = [];
    var checkboxes = document.querySelectorAll('input[type="checkbox"]');
    for(var i=0; i<checkboxes.length; i++){
        if(checkboxes[i].checked){
            selected.push(checkboxes[i].parentNode.textContent.trim());
        }
    }

    if(selected.length>0){
        alert('You chose: ' + selected.join(', '));
    } else {
        alert('No items picked üòÖ');
    }
}
